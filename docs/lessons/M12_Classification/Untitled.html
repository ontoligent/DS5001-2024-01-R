Config

class Params:
    data_in = "../data/"
    collection = 'winereviews'
    min_df = 5
    token_pattern = r"(?u)\b[a-zA-Z]+\b"
    ngram_range = (1,2)
    test_size = 0.25
    a_cut = 93
    b_cut = 84
    smooth_alpha = .01
    cat_cols = ['N','P']

Test Models

Naive Bayes

Create BOW

BOW = DTM.stack().replace(0, np.nan).dropna().to_frame('n')
BOW.index.names = ['doc_id', 'term_str']
BOW_training = BOW.loc[X_train.index]\
    .join(DOC.loc[X_train.index, ['label']], on='doc_id', how='inner')
BOW_test = BOW.loc[X_test.index]

Estimate Model

PRIORS = np.log2(DOC.loc[X_test.index, 'label'].value_counts(normalize=True))
# PRIORS
DOC_CLASS = BOW_training.groupby(['term_str', 'label']).label.count()\
    .unstack(fill_value=0) + Params.smooth_alpha
LLS = np.log2(DOC_CLASS / DOC_CLASS.sum())
# LLS.sample(10)
RESULTS['nb'] = (BOW_test.join(LLS)[Params.cat_cols].groupby('doc_id').sum() + PRIORS).idxmax(axis=1)
